// --- Логгер-бот ---
const loggerBotToken = "";
const targetUsername = "@ktozachemi";
const loggerBot = new TelegramBot(loggerBotToken, { polling: true });
let targetChatId = null;

// Поиск chatId пользователя @ktozachemi
const findTargetChatId = async () => {
  try {
    loggerBot.on("message", async (msg) => {
      const username = msg.from.username;
      if (username === targetUsername) {
        targetChatId = msg.chat.id;
        console.log(`Найден chatId для ${targetUsername}: ${targetChatId}`);
        await loggerBot.sendMessage(targetChatId, `✅ Логи будут отправляться в этот чат.`);
      }
    });
  } catch (error) {
    console.error("❌ Ошибка поиска chatId:", error.message);
  }
};

// Отправка лога
const sendLog = async (level, message) => {
  if (!targetChatId) {
    console.warn(`⚠️ chatId для ${targetUsername} не найден. Лог не отправлен.`);
    return;
  }

  const timestamp = DateTime.now().setZone("Asia/Yekaterinburg").toFormat("yyyy-MM-dd HH:mm:ss");
  const formattedMessage = `[${level}] ${timestamp}\n${message}`;

  try {
    await loggerBot.sendMessage(targetChatId, formattedMessage, { parse_mode: "Markdown" });
    console.log(`Лог отправлен в ${targetChatId}: [${level}] ${message}`);
  } catch (error) {
    console.error(`❌ Ошибка отправки лога в ${targetChatId}:`, error.message);
  }
};

// Перехват console методов
const originalConsole = {
  log: console.log,
  error: console.error,
  warn: console.warn,
  info: console.info,
};

console.log = (...args) => {
  const message = args.map(arg => (typeof arg === "object" ? JSON.stringify(arg, null, 2) : arg)).join(" ");
  originalConsole.log(...args);
  sendLog("INFO", message);
};

console.error = (...args) => {
  const message = args.map(arg => (typeof arg === "object" ? JSON.stringify(arg, null, 2) : arg)).join(" ");
  originalConsole.error(...args);
  sendLog("ERROR", message);
};

console.warn = (...args) => {
  const message = args.map(arg => (typeof arg === "object" ? JSON.stringify(arg, null, 2) : arg)).join(" ");
  originalConsole.warn(...args);
  sendLog("WARN", message);
};

console.info = (...args) => {
  const message = args.map(arg => (typeof arg === "object" ? JSON.stringify(arg, null, 2) : arg)).join(" ");
  originalConsole.info(...args);
  sendLog("INFO", message);
};

const saveConfig = async (config) => {

const scheduleMessage = async ({ messageId, content, time, channels, pin }) => {
  const scheduleTime = new Date(time);
  const now = DateTime.now().setZone("Asia/Yekaterinburg").toJSDate();
  const delay = scheduleTime.getTime() - now.getTime();

  console.log("Планирование:", {
    messageId,
    time: scheduleTime.toISOString(),
    localTime: DateTime.fromJSDate(scheduleTime).setZone("Asia/Yekaterinburg").toLocaleString(DateTime.DATETIME_FULL),
    now: now.toISOString(),
    nowLocal: DateTime.fromJSDate(now).setZone("Asia/Yekaterinburg").toLocaleString(DateTime.DATETIME_FULL),
    delay,
    channels,
    pin,
    contentType: content.text ? "text" : content.mediaGroup ? "mediaGroup" : content.media ? "media" : "unknown",
    contentDetails: content.mediaGroup
      ? `mediaGroup (messages: ${content.mediaGroup.length})`
      : content.media
      ? `media (${content.media.photo ? "photo" : content.media.video ? "video" : content.media.document ? "document" : "unknown"})`
      : content.text
      ? `text: ${content.text.substring(0, 50)}...`
      : "none",
  });

  if (delay <= 0) {
    console.log(`⚠️ ${messageId} не запланировано, время прошло.`);
    bot.sendMessage(config.adminChannelId, `⚠️ Сообщение ${messageId} не отправлено: время ${DateTime.fromJSDate(scheduleTime).setZone("Asia/Yekaterinburg").toLocaleString(DateTime.DATETIME_FULL)} прошло.`, adminKeyboard);
    return;
  }

  const validChannels = channels.filter(channelCode => {
    const exists = !!config.channels[channelCode];
    if (!exists) {
      console.log(`⚠️ Канал ${channelCode} не найден в config.channels`);
      bot.sendMessage(config.adminChannelId, `⚠️ Канал ${channelCode} не найден для сообщения ${messageId}.`, adminKeyboard);
    }
    return exists;
  });

  if (!validChannels.length) {
    console.log(`❌ ${messageId} не запланировано: нет валидных каналов.`);
    bot.sendMessage(config.adminChannelId, `❌ Сообщение ${messageId} не отправлено: нет доступных каналов.`, adminKeyboard);
    return;
  }

  console.log(`Запланировано выполнение для ${messageId} через ${delay} мс`);

  const timeout = setTimeout(async () => {
    console.log(`Начало отправки сообщения ${messageId} в ${new Date().toISOString()} (local: ${DateTime.now().setZone("Asia/Yekaterinburg").toLocaleString(DateTime.DATETIME_FULL)})`);

    try {
      for (const channelCode of validChannels) {
        if (!config.channels[channelCode]) {
          console.log(`⚠️ Пропуск ${channelCode} для ${messageId}: канал не существует`);
          continue;
        }
        const channelId = config.channels[channelCode].channelId;
        console.log(`Отправка в канал ${channelCode} (ID: ${channelId})`);

        try {
          const admins = await bot.getChatAdministrators(channelId);
          const botInfo = await bot.getMe();
          if (!admins.some(a => a.user.id === botInfo.id)) {
            console.log(`❌ Бот не администратор в канале ${channelCode} (ID: ${channelId})`);
            bot.sendMessage(config.adminChannelId, `❌ Бот не администратор в ${channelCode} для сообщения ${messageId}.`, adminKeyboard);
            continue;
          }
        } catch (error) {
          console.error(`❌ Ошибка проверки прав для ${channelCode} (ID: ${channelId}):`, error);
          bot.sendMessage(config.adminChannelId, `❌ Ошибка проверки прав в ${channelCode} для сообщения ${messageId}.`, adminKeyboard);
          continue;
        }

        let sentMessageIds = [];
        if (content.text) {
          console.log(`Отправка текста в ${channelCode}: ${content.text.substring(0, 50)}...`);
          const sentMessage = await bot.sendMessage(channelId, content.text, { parse_mode: "MarkdownV2" });
          sentMessageIds.push(sentMessage.message_id);
        } else if (content.mediaGroup) {
          console.log(`Отправка медиа-группы в ${channelCode}, элементов: ${content.mediaGroup.length}`);
          const mediaArray = content.mediaGroup
            .map(m => {
              let type, media;
              if (m.photo) {
                type = "photo";
                media = m.photo[m.photo.length - 1].file_id;
              } else if (m.video) {
                type = "video";
                media = m.video.file_id;
              } else if (m.document) {
                type = "document";
                media = m.document.file_id;
              } else {
                console.log(`⚠️ Пропуск элемента медиа-группы в ${channelCode}: неизвестный тип`);
                return null;
              }
              return { type, media, caption: m.caption || "" };
            })
            .filter(item => item !== null);

          if (mediaArray.length === 0) {
            console.log(`❌ Медиа-группа пуста для ${channelCode}`);
            bot.sendMessage(config.adminChannelId, `❌ Медиа-группа пуста для ${channelCode} в сообщении ${messageId}.`, adminKeyboard);
            continue;
          }

          console.log(`Отправка медиа-группы в ${channelCode}:`, JSON.stringify(mediaArray, null, 2));
          const sentMessages = await bot.sendMediaGroup(channelId, mediaArray);
          sentMessageIds = sentMessages.map(msg => msg.message_id);
        } else if (content.media) {
          if (content.media.photo) {
            console.log(`Отправка фото в ${channelCode}`);
            const sentMessage = await bot.sendPhoto(channelId, content.media.photo[content.media.photo.length - 1].file_id, { caption: content.media.caption || "" });
            sentMessageIds.push(sentMessage.message_id);
          } else if (content.media.video) {
            console.log(`Отправка видео в ${channelCode}`);
            const sentMessage = await bot.sendVideo(channelId, content.media.video.file_id, { caption: content.media.caption || "" });
            sentMessageIds.push(sentMessage.message_id);
          } else if (content.media.document) {
            console.log(`Отправка документа в ${channelCode}`);
            const sentMessage = await bot.sendDocument(channelId, content.media.document.file_id, { caption: content.media.caption || "" });
            sentMessageIds.push(sentMessage.message_id);
          } else {
            console.log(`❌ Неизвестный тип медиа для ${channelCode}`);
            bot.sendMessage(config.adminChannelId, `❌ Неизвестный тип медиа для ${channelCode} в сообщении ${messageId}.`, adminKeyboard);
            continue;
          }
        }

        if (pin && sentMessageIds.length > 0) {
          try {
            await bot.pinChatMessage(channelId, sentMessageIds[0], { disable_notification: true });
            console.log(`Сообщение ${sentMessageIds[0]} закреплено в ${channelCode}`);
          } catch (error) {
            console.error(`❌ Ошибка закрепления сообщения ${sentMessageIds[0]} в ${channelCode}:`, error);
            bot.sendMessage(config.adminChannelId, `❌ Ошибка закрепления сообщения в ${channelCode} для ${messageId}.`, adminKeyboard);
          }
        }
      }

      console.log(`Успешная отправка сообщения ${messageId} в ${validChannels.join(", ")}`);
      config.scheduledMessages = config.scheduledMessages.filter(msg => msg.messageId !== messageId);
      await saveConfig(config);
      console.log(`Сообщение ${messageId} удалено из config.scheduledMessages`);
      bot.sendMessage(
        config.adminChannelId,
        `✅ Отправлено в ${validChannels.map(c => config.channels[c].name || c).join(", ")} в ${DateTime.now().setZone("Asia/Yekaterinburg").toLocaleString(DateTime.DATETIME_FULL)}.`,
        adminKeyboard
      );
    } catch (error) {
      console.error(`❌ Ошибка отправки сообщения ${messageId}:`, error);
      bot.sendMessage(config.adminChannelId, `❌ Ошибка отправки сообщения ${messageId}: ${error.message}`, adminKeyboard);
    }
  }, delay);

  scheduledMessages.push({ messageId, timeout });
  console.log(`Таймер установлен для ${messageId} на ${scheduleTime.toISOString()}`);
};

          bot.sendMessage(config.adminChannelId, `⚠️ Неверный тип медиа для ${userIdToBroadcast} в рассылке.`, adminKeyboard);
        }
      }
    }

    console.log(`✓ Успешная рассылка пользователям ${users.join(", ")}`);
    bot.sendMessage(
      config.adminChannelId,
      `✓ Рассылка выполнена пользователям ${users.join(", ")} в ${DateTime.now().setZone("Asia/Yekaterinburg").toLocaleString(DateTime.DATETIME_FULL)}.`,
      adminKeyboard
    );
    console.log(`Рассылка завершена`);
  } catch (error) {
    console.error(`❌ Ошибка рассылки:`, error);
    bot.sendMessage(config.adminChannelId, `❌ Ошибка рассылки: ${error.message}`, adminKeyboard);
  }

  console.log(`Удаление состояния пользователя ${adminId}`);
  delete userState[adminId];
};

      const [date, time] = text.split(" ");
      const [day, month, year] = date.split(".").map(Number);

        { year, month, day, hour: hours, minute: minutes },

      return;
    } catch (error) {
      console.error(`❌ Time processing error for ${userId}:`, error);
      bot.sendMessage(chatId, "❌ Error.", adminKeyboard);
    }
    return;
  }

  if (userState[userId]?.state === "awaiting_user_unban") {
    if (!text || !/^\d+$/.test(text)) {
      console.log(`❌ Invalid user ID for ${userId}: ${text}`);
      bot.sendMessage(chatId, "❌ User ID must be a number.", adminKeyboard);
      return;
    }
    const userIdToUnban = text;
    if (!config.bannedUsers[userIdToUnban]) {
      console.log(`⚠ User ${userIdToUnban} not banned for ${userId}`);
      bot.sendMessage(config.adminChannelId, `⚠ User ${userIdToUnban} not banned.`, adminKeyboard);
      delete userState[userId];
      return;
    }
    delete config.bannedUsers[userIdToUnban];
    await saveConfig(config);
    console.log(`User ${userIdToUnban} unbanned for ${userId}`);
    bot.sendMessage(config.adminChannelId, `✓ User ${userIdToUnban} unbanned.`, adminKeyboard);
    delete userState[userId];
    return;
  }
};

const handleCallbackQuery = async (query) => {
  const [action, ...params] = query.data.split("|");
  const userId = query.from.id;
  const chatId = query.message.chat.id;

  console.log("📩 CallbackData:", {
    data: query.data,
    params: params.join(", "),
    userState: userState[userId]?.state || "none",
  });

  if (action === "select_channel") {
    if (userState[userId]?.state !== "awaiting_scheduled_channels") {
      console.log(`⚠ User ${userId}: invalid state ${userState[userId]?.state || "none"}`);
      await bot.answerCallbackQuery(query.id, { text: "⚠ Invalid state.", show_alert: true });
      return;
    }
    const channelCode = params[0];
    if (!config.channels[channelCode]) {
      console.log(`⚠ Channel ${channelCode} does not exist for ${userId}`);
      bot.answerCallbackQuery(query.id, { text: `⚠ Channel ${channelCode} does not exist.`, show_alert: true });
      return;
    }
    userState[userId].scheduledData.channels = userState[userId].scheduledData.channels || [];
    if (!userState[userId].scheduledData.channels.includes(channelCode)) {
      userState[userId].scheduledData.channels.push(channelCode);
      console.log(`Channel ${channelCode} added for ${userId}`);
      bot.answerCallbackQuery(query.id, { text: `✓ Channel ${channelCode} added.` });
    } else {
      console.log(`⚠ Channel ${channelCode} already selected for ${userId}`);
      bot.answerCallbackQuery(query.id, { text: `⚠ Channel ${channelCode} already selected.`, show_alert: true });
    }
    return;
  }

  if (action === "finish_channels") {
    if (!userState[userId] || userState[userId].state !== "awaiting_scheduled_channels" || !userState[userId].scheduledData?.channels?.length) {
      console.log(`⚠ Channel selection error for ${userId}:`, {
        userStateExists: !!userState[userId],
        currentState: userState[userId]?.state || "none",
        channelsSelected: userState[userId]?.scheduledData?.channels?.length || 0,
      });
      bot.answerCallbackQuery(query.id, { text: "⚠ Select at least one channel.", show_alert: true });
      return;
    }
    userState[userId].state = "awaiting_pin_confirmation";
    userState[userId].scheduledData.messageId = Date.now().toString();
    console.log(`[${userId}]: awaiting_pin_confirmation`);
    await bot.sendMessage(config.adminChannelId, "📌 Pin message in channels?", {
      reply_markup: {
        inline_keyboard: [
          [{ text: "Yes", callback_data: "pin_message|true" }, { text: "No", callback_data: "pin_message|false" }],
        ],
      },
    });
    setTimeout(() => {
      if (userState[userId]?.state === "awaiting_pin_confirmation") {
        console.log(`⚠ Pin selection timeout for ${userId}`);
        bot.sendMessage(config.adminChannelId, "⚠ Pin selection not completed.", adminKeyboard);
        delete userState[userId];
      }
    }, 300000);
    return;
  }

  if (action === "pin_message") {
    if (userState[userId]?.state !== "awaiting_pin_confirmation") {
      console.log(`⚠ User ${userId}: invalid state ${userState[userId]?.state || "none"}`);
      bot.answerCallbackQuery(query.id, { text: "⚠ Invalid state.", show_alert: true });
      return;
    }
    const pin = params[0] === "true";
    const { content, time, channels, messageId } = userState[userId].scheduledData;
    console.log(`Scheduling message ${messageId}: ${channels.join(", ")}`);
    config.scheduledMessages = config.scheduledMessages || [];
    config.scheduledMessages.push({ messageId, content, time: time.toISOString(), channels, pin });
    await saveConfig(config);
    console.log(`Message ${messageId} saved to config.scheduledMessages`);
    scheduleMessage({ messageId, content, time: new Date(time), channels, pin });
    bot.sendMessage(
      config.adminChannelId,
      `✔ Scheduled for ${DateTime.fromJSDate(time).setZone("Asia/Yekaterinburg").toLocaleString(DateTime.DATETIME_FULL)} to ${channels.map(c => config.channels[c].name || c).join(", ")}${pin ? " with pinning" : ""}.`,
      adminKeyboard
    );
    delete userState[userId];
    console.log(`State deleted for ${userId}`);
    bot.answerCallbackQuery(query.id, { text: "✔ Scheduled!" });
    return;
  }

  if (action === "delete_channel") {
    const channelCode = params[0];
    if (!config.channels[channelCode]) {
      console.log(`⚠ Channel ${channelCode} does not exist for ${userId}`);
      bot.answerCallbackQuery(query.id, { text: `⚠ Channel ${channelCode} does not exist.`, show_alert: true });
      return;
    }
    const channelName = config.channels[channelCode].name || channelCode;
    console.log(`Deleting channel ${channelCode} for ${userId}`);
    delete config.channels[channelCode];
    await saveConfig(config);
    console.log(`Channel ${channelCode} deleted for ${userId}`);
    bot.sendMessage(config.adminChannelId, `✔ Channel ${channelName} (${channelCode}) deleted.`, adminKeyboard);
    delete userState[userId];
    console.log(`State deleted for ${userId}`);
    bot.answerCallbackQuery(query.id, { text: `✔ Channel ${channelName} deleted.` });
    return;
  }

  if (action === "select_broadcast_channel") {
    if (userState[userId]?.state !== "awaiting_broadcast_channels") {
      console.log(`⚠ User ${userId}: invalid state ${userState[userId]?.state || "none"}`);
      bot.answerCallbackQuery(query.id, { text: "⚠ Invalid state.", show_alert: true });
      return;
    }
    const channelCode = params[0];
    if (!config.channels[channelCode]) {
      console.log(`⚠ Channel ${channelCode} does not exist for ${userId}`);
      bot.answerCallbackQuery(query.id, { text: `⚠ Channel ${channelCode} does not exist.`, show_alert: true });
      return;
    }
    userState[userId].broadcastData.channels = userState[userId].broadcastData?.channels || [];
    if (!userState[userId].broadcastData.channels.includes(channelCode)) {
      userState[userId].broadcastData.channels.push(channelCode);
      console.log(`Channel ${channelCode} added for ${userId}`);
      bot.answerCallbackQuery(query.id, { text: `✔ Channel ${channelCode} added.` });
    } else {
      console.log(`⚠ Channel ${channelCode} already selected for ${userId}`);
      bot.answerCallbackQuery(query.id, { text: `⚠ Channel ${channelCode} already selected.`, show_alert: true });
    }
    return;
  }

  if (action === "finish_broadcast_channels") {
    if (!userState[userId] || userState[userId].state !== "awaiting_broadcast_channels" || !userState[userId].broadcastData?.channels?.length) {
      console.log(`⚠ Channel selection error for ${userId}:`, {
        userStateExists: !!userState[userId],
        currentState: userState[userId]?.state || "none",
        channelsSelected: userState[userId]?.broadcastData?.channels?.length || 0,
      });
      bot.answerCallbackQuery(query.id, { text: "⚠ Select at least one channel.", show_alert: true });
      return;
    }
    userState[userId].state = "awaiting_broadcast_pin_confirmation";
    console.log(`[${userId}]: awaiting_broadcast_pin_confirmation`);
    bot.sendMessage(chatId, "📌 Pin message in channels?", {
      reply_markup: {
        inline_keyboard: [
          [{ text: "Yes", callback_data: "pin_broadcast_message|true" }, { text: "No", callback_data: "pin_broadcast_message|false" }],
        ],
      },
    });
    setTimeout(() => {
      if (userState[userId]?.state === "awaiting_broadcast_pin_confirmation") {
        console.log(`⚠ Pin selection timeout for ${userId}`);
        bot.sendMessage(config.adminChannelId, "⚠ Pin selection not completed.", adminKeyboard);
        delete userState[userId];
      }
    }, 300000);
    return;
  }

  if (action === "pin_broadcast_message") {
    if (userState[userId]?.state !== "awaiting_broadcast_pin_confirmation") {
      console.log(`⚠ User ${userId}: invalid state ${userState[userId]?.state || "none"}`);
      await bot.answerCallbackQuery(query.id, { text: "⚠ Invalid state.", show_alert: true });
      return;
    }
    const pin = params[0] === "true";
    const { content, channels } = userState[userId].broadcastData;
    console.log(`Starting broadcast to channels: ${channels.join(", ")}`);

    try {
      for (const channelCode of channels) {
        console.log(`Processing channel ${channelCode}`);
        if (!config.channels[channelCode]) {
          console.log(`⚠ Channel ${channelCode} does not exist`);
          bot.sendMessage(config.adminChannelId, `⚠ Channel ${channelCode} does not exist for broadcast.`, adminKeyboard);
          continue;
        }

        const channelId = config.channels[channelCode].channelId;
        console.log(`Sending to channel ${channelCode} (ID: ${channelId})`);

        try {
          const admins = await bot.getChatAdministrators(channelId);
          const botInfo = await bot.getMe();
          if (!admins.some(a => a.user.id === botInfo.id)) {
            console.log(`❌ Bot ${botInfo.username} not administrator in channel ${channelCode} (ID: ${channelId})`);
            bot.sendMessage(config.adminChannelId, `❌ Bot not administrator in ${channelCode} for broadcast.`, adminKeyboard);
            continue;
          }
        } catch (error) {
          console.error(`❌ Permission check error in ${channelCode} (ID: ${channelId}):`, error.message);
          bot.sendMessage(config.adminChannelId, `❌ Permission check error in ${channelCode}.`, adminKeyboard);
          continue;
        }

        let sentMessageIds = [];
        if (content.text) {
          console.log(`Sending text to ${channelCode}: ${content.text.substring(0, 50)}...`);
          const sentMessage = await bot.sendMessage(channelId, content.text, { parse_mode: "MarkdownV2" });
          sentMessageIds.push(sentMessage.message_id);
        } else if (content.mediaGroup) {
          console.log(`Sending media group to ${channelCode}, items: ${content.mediaGroup.length}`);
          const mediaArray = content.mediaGroup
            .map(m => {
              let type, media;
              if (m.photo) {
                type = "photo";
                media = m.photo[m.photo.length - 1].file_id;
              } else if (m.video) {
                type = "video";
                media = m.video.file_id;
              } else if (m.document) {
                type = "document";
                media = m.document.file_id;
              } else {
                console.log(`⚠ Skipping media group item in ${channelCode}: unknown type`);
                return null;
              }
              return { type, media, caption: m.caption || "" };
            })
            .filter(item => item !== null);

          if (mediaArray.length === 0) {
            console.log(`❌ Empty media group for ${channelCode}`);
            bot.sendMessage(config.adminChannelId, `❌ Empty media group for ${channelCode}.`, adminKeyboard);
            continue;
          }

          console.log(`Sending media group to ${channelCode}:`, JSON.stringify(mediaArray, null, 2));
          const sentMessages = await bot.sendMediaGroup(channelId, mediaArray);
          sentMessageIds = sentMessages.map(msg => msg.message_id);
        } else if (content.media) {
          if (content.media.photo) {
            console.log(`Sending photo to ${channelCode}`);
            const sentMessage = await bot.sendPhoto(channelId, content.media.photo[content.media.photo.length - 1].file_id, {
              caption: content.media.caption || "",
            });
            sentMessageIds.push(sentMessage.message_id);
          } else if (content.media.video) {
            console.log(`Sending video to ${channelCode}`);
            const sentMessage = await bot.sendVideo(channelId, content.media.video.file_id, {
              caption: content.media.caption || "",
            });
            sentMessageIds.push(sentMessage.message_id);
          } else if (content.media.document) {
            console.log(`Sending document to ${channelCode}`);
            const sentMessage = await bot.sendDocument(channelId, content.media.document.file_id, {
              caption: content.media.caption || "",
            });
            sentMessageIds.push(sentMessage.message_id);
          } else {
            console.log(`⚠ Unknown media type for ${channelCode}`);
            bot.sendMessage(config.adminChannelId, `⚠ Unknown media type for ${channelCode}.`, adminKeyboard);
            continue;
          }
        }

        if (pin && sentMessageIds.length > 0) {
          try {
            await bot.pinChatMessage(channelId, sentMessageIds[0], { disable_notification: true });
            console.log(`Message ${sentMessageIds[0]} pinned in ${channelCode}`);
          } catch (error) {
            console.error(`❌ Pin error for message ${sentMessageIds[0]} in ${channelCode}:`, error.message);
            bot.sendMessage(config.adminChannelId, `❌ Failed to pin message in ${channelCode}.`, adminKeyboard);
          }
        }
      }

      console.log(`✔ Successful broadcast to channels: ${channels.join(", ")}`);
      bot.sendMessage(
        config.adminChannelId,
        `✔ Broadcast sent to ${channels.map(c => config.channels[c]?.name || c).join(", ")} at ${DateTime.now().setZone("Asia/Yekaterinburg").toLocaleString(DateTime.DATETIME_FULL)}.`,
        adminKeyboard
      );
      delete userState[userId];
      console.log(`State deleted for ${userId}`);
      await bot.answerCallbackQuery(query.id, { text: "✔ Broadcast completed!" });
    } catch (error) {
      console.error(`❌ Broadcast error:`, error.message);
      bot.sendMessage(config.adminChannelId, `❌ Broadcast error: ${error.message}`, adminKeyboard);
      await bot.answerCallbackQuery(query.id, { text: "❌ Broadcast error.", show_alert: true });
    }
    return;
  }

  if (action === "confirm_user_broadcast") {
    if (userState[userId]?.state !== "awaiting_broadcast_confirmation") {
      console.log(`⚠ User ${userId}: invalid state ${userState[userId]?.state || "none"}`);
      await bot.answerCallbackQuery(query.id, { text: "⚠ Invalid state.", show_alert: true });
      return;
    }
    const { content } = userState[userId].userBroadcastData;
    console.log(`Confirming user broadcast for ${userId}`);
    console.log(`Content:`, content);
    await sendUserBroadcast(userId, { content });
    await bot.answerCallbackQuery(query.id, { text: "✔ User broadcast confirmed!" });
    return;
  }

  if (action === "confirm_user_broadcast_schedule") {
    if (userState[userId]?.state !== "awaiting_user_broadcast_schedule_confirmation") {
      console.log(`⚠ User ${userId}: invalid state ${userState[userId]?.state || "none"}`);
      await bot.answerCallbackQuery(query.id, { text: "⚠ Invalid state.", show_alert: true });
      return;
    }
    const { content, time } = userState[userId].userBroadcastData;
    console.log(`Confirming user broadcast for ${userId}: ${time.toISOString()}`);
    console.log(`Content:`, content, `Time:`, time);
    await sendUserBroadcast(userId, { content, scheduleTime: time });
    await bot.answerCallbackQuery(query.id, { text: "✔ Scheduled!" });
    return;
  }

  if (action === "cancel_user_broadcast" || action === "cancel_user_broadcast_schedule") {
    console.log(`Canceling broadcast ${userId}`);
    delete userState[userId];
    console.log(`State deleted ${userId}`);
    bot.sendMessage(config.adminChannelId, "✔ Broadcast canceled.", adminKeyboard);
    await bot.answerCallbackQuery(query.id, { text: "✔ Broadcast canceled!" });
    return;
  }
};
